```typescript
export async function PATCH(request: Request) {
  try {
    const body = await request.json();
    const { updates } = bulkCategorizeSchema.parse(body);

    // Empty-input early return — no SQL issued. Cheap defensive path.
    if (updates.length === 0) {
      return success({ updated: 0, total: 0 });
    }

    // Build parallel arrays for the unnest() driver. Order is preserved
    // between ids[i] and closenesses[i] — same index = same update pair.
    const ids = updates.map((u) => u.contactId);
    const closenesses = updates.map((u) => u.closeness);

    // Single round-trip bulk UPDATE.
    // CLAUDE.md "no raw SQL" guard: `sql` template tag inside db.execute()
    // is the documented Drizzle escape for batched UPDATE/INSERT (D-06).
    // Parameter binding: ${ids} and ${closenesses} are interpolated as
    // bound parameters (NOT string-concatenated), preserving SQL safety.
    // RETURNING id gives the count of actually-mutated rows for the response.
    //
    // Identifiers (`contacts`, `contacts.id`) are LITERAL in the template —
    // not `${contacts}` / `${contacts.id}` pgTable interpolations. The
    // literal form has analog precedent in metrics/dashboard/route.ts and
    // avoids relying on Drizzle v0.45.1 pgTable-interpolation behavior that
    // has no repo precedent (WARNING 2 fix). Trade-off: a future `contacts`
    // table rename would require updating this SQL by hand — accepted.
    const result = await db.execute(sql`
      UPDATE contacts
      SET closeness = data.cl,
          updated_at = NOW()
      FROM (
        SELECT * FROM unnest(${ids}::uuid[], ${closenesses}::contact_closeness[])
        AS d(id, cl)
      ) AS data
      WHERE contacts.id = data.id
      RETURNING id
    `);

    const updated = (result.rows as Array<{ id: string }>).length;
    return success({ updated, total: updates.length });
  } catch (err) {
    if (err instanceof z.ZodError) return validationError(err.issues[0].message);
    return serverError(err);
  }
}
```
