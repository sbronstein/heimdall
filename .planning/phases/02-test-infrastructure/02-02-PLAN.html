From src/features/job-leads/lib/prioritization.ts (full formula reproduced for test fidelity):

closenessWeights = { close_friend: 100, close_colleague: 90, friend: 75, colleague: 60,
                     career_contact: 45, acquaintance: 30, linkedin_only: 15, never_met: 5 };
recencyWeight(lastContactDate) = lastContactDate ? max(0, 100 - daysSince/3.65) : 0;
computeBridgeScore(prospect, contact) =
  Math.round(0.4 * seniorityWeights[prospect.seniorityLevel] +
             0.35 * closenessWeights[contact.closeness ?? 'acquaintance'] +
             0.25 * recencyWeight(contact.lastContactDate));
