```bash
# Drain mode
TOKEN=$(cat ~/.heimdall/api-token)
curl -s -H "Authorization: Bearer $TOKEN" "http://localhost:4000/api/job-leads?status=queued&limit=50"

# Claim

  -d '{"status":"searching"}' "http://localhost:4000/api/job-leads/$LEAD_ID/status"

# Write success
curl -s -X POST -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" \
  -d "{\"prospects\":$JSON}" "http://localhost:4000/api/job-leads/$LEAD_ID/prospects"

# Write failure

  "http://localhost:4000/api/job-leads/$LEAD_ID/status"
```
